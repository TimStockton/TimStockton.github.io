#!/bin/sh

# script name: my_which
# script author: Timothy Stockton
# last modified: 20251008

# this script determines if the provided arguments are executable files
# it is built for portability
# option flags include the following:
#    -a : find all occurances, not just the first

# helper functions

# determines presence of "/" in filename
is_pathname() {
    string=$1

    if [ "$string" = */* ]; then
        return 0 # true
    else
        return 1 # false
    fi
}

# determines if a file is executable
# also ensures the file exists and is not a directory
is_executable_file() {
    filepath=$1

    if [ -e "$filepath" ] && [ -x "$filepath" ] && [ ! -d "$filepath" ]; then
        return 0   # true
    else
        return 1   # false
    fi
}

# adds . for the current directory to colons in PATH
process_path() {
    # save the original PATH and field separator
    original_path=$PATH
    OLDIFS=$IFS

    # handle special PATH cases before splitting

    # leading colon
    # add current directory before
    if [ "$original_path" = :* ]; then
        original_path=".$original_path"
    fi

    # double colon
    # add current directory between
    original_path=$(echo "$original_path" | sed 's/::/:.:/g')

    # trailing colon
    # add current directory after
    if [ "$original_path" = *: ];then
        original_path="${original_path}."
    fi

    # split the processed PATH by colon
    IFS=':'
    for dir in $original_path; do
        # output each directory in newline seperated list
        # used to iterate through directories in a loop
        echo "$dir"
    done

    # restore IFS
    IFS=$OLDIFS
}

# prepare -a option flag
FINDALL=FALSE

# check first parameter for -a option
if [ "$1" = "-a" ]; then
    # update flag
    FINDALL=TRUE
    # dispose of $1 so it is not checked for executability
    shift
fi

# Loop over all remaining parameters ($@)
# assigns each parameter to the variable $file
for file in "$@"; do
    if is_pathname "$file"; then
        if is_executable_file "$file"; then
            # output executable file
            echo "$file"
        else
            echo "$file NOT FOUND"
        fi
    else
        # not a pathname
        # prep flag
        FOUND=FALSE

        # add colons to $PATH, split on colons, iterate
        # assigns each directory to the variable $P
        for P in $(process_path); do
            # if $P/$file is executable and not a directory
            if is_executable_file "$P/$file"; then
                # update flag
                FOUND=TRUE
                # output executable file
                echo "$P/$file"
                if [ "$FINDALL" = "FALSE" ]; then
                    # -a not present, exit the for loop
                    break
                fi
            fi
        done
        if [ "$FOUND" = "FALSE" ]; then
            echo "$file NOT FOUND"
        fi
    fi
    # done checking current file, remove it from arg list
    shift
done
